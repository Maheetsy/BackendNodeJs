const express = require('express');
const router = express.Router();
const {
  registerUser,
  getAllUsers,
  getUserById,
  updateUser,
  deleteUser
} = require('../controllers/user.controller');
const { protect, authorize } = require('../middleware/auth');

// --- RUTA PÚBLICA ---
// Esta ruta es para que cualquiera pueda registrarse.
// No necesita 'protect' ni 'authorize'.
router.post('/register', registerUser);

// --- RUTAS PROTEGIDAS ---
// A partir de aquí, TODAS las rutas requieren un token (protect)
// y ser admin/gerente (authorize).
router.use(protect);
router.use(authorize('admin', 'gerente'));

// Estas rutas SÍ están protegidas:
router.get('/', getAllUsers);
router.get('/:id', getUserById);
router.put('/:id', updateUser);
router.delete('/:id', deleteUser);

module.exports = router;
